\begin{tehtava}
    Olkoot $A$: ''Matti on insinööri'' ja $B$: ''Matilla on hyvä työpaikka''.
Suomenna lause.
    \begin{alakohdat}
        \alakohta{$A\land B$,}
        \alakohta{$\lnot A \lor B$,}
        \alakohta{$A\to B$,}
        \alakohta{$\lnot B\to A$,}
        \alakohta{$\lnot( B \to A )$ ja}
        \alakohta{$ B \lequiv A$.}
    \end{alakohdat}

    \begin{vastaus}
        \begin{alakohdat}
            \alakohta{Matti on insinööri, jolla on hyvä työpaikka.}
            \alakohta{Matti ei ole insinööri tai hänellä on hyvä työpaikka.}
            \alakohta{Jos Matti on insinööri, hänellä on hyvä työpaikka.}
            \alakohta{Jos Matilla ei ole hyvää työpaikkaa, hän on insinööri.}
            \alakohta{Ei ole totta, että jos Matilla on hyvä työpaikka, hän olisi insinööri.}
            \alakohta{Matilla on hyvä työpaikka, jos ja vain jos hän on insinööri.}
        \end{alakohdat}
    \end{vastaus}
    
\end{tehtava}
