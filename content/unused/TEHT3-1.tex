\begin{tehtava}

        \alakohta{$\compl A$,}

        \alakohta{$A\setminus B$,}

    \end{alakohdat}

    \begin{vastaus}
    
        \begin{alakohdat}
    Perusjoukko on koulun opiskelijat.
            \alakohta{Vanhempiensa luona asuvat täysi-ikäiset.}

            \alakohta{Opiskelijat, jotka eivät ole täysi-ikäisiä eivätkä asu vanhempiensa luona.}
        \end{alakohdat}
    \end{vastaus}
    
\end{tehtava}
