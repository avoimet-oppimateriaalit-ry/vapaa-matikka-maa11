\begin{tehtava}

	+$T(x)$: ''opiskelija $x$ on täysi-ikäinen'' ja
	+perusjoukko kaikki koulun opiskelijat. Suomenna lause.
	\begin{alakohdat}
	\alakohta{$\forall x T(x)$}
	\alakohta{$\exists x T(x)$}
	\alakohta{$\exists x \lnot T(x)$}
	\alakohta{$\forall x \lnot T(x)$}
	\end{alakohdat}%
	\begin{vastaus}
		\begin{alakohdat}
			\alakohta{Kaikki koulun opiskelijat ovat täysi-ikäisiä.}
			\alakohta{Ainakin yksi koulun opiskelija on täysi-ikäinen.}
			\alakohta{Ainakin yksi koulun opiskelija ei ole täysi-ikäinen.}
			\alakohta{Yksikään koulun opiskelija ei ole täysi-ikäinen.}
		\end{alakohdat}
	\end{vastaus}
\end{tehtava}

\begin{tehtava}
	Suomenna lause. Onko lause tosi? Perustele.
	\begin{alakohdat}
	\alakohta{$\forall x\in\rr (|x| > 0)$}
	\alakohta{$\forall x\in\rr (|x| \ge 0)$}
	\alakohta{$\exists x\in\nn (x^2 < 2)$}
	\alakohta{$\exists x\in\nn (x^2 = 2)$}
	\end{alakohdat}%
	\begin{vastaus}
		\begin{alakohdat}
			\alakohta{Lause on epätosi, sillä $0 \in \mathbb{R}$ ja $|0| = 0$.}
			\alakohta{Lause on tosi, sillä reaaliluvun itseisarvo on aina suurempaa tai yhtä suurta kuin $0$.}
			\alakohta{Lause on tosi, sillä $1 \in \mathbb{N}$ ja $1^2 < 2$.}
			\alakohta{Lause on epätosi, sillä yhtälön $x^2 = 2$ ratkaisut $\sqrt{2}$ ja $-\sqrt{2}$ eivät ole luonnollisia lukuja.}
		\end{alakohdat}
	\end{vastaus}
\end{tehtava}

\begin{tehtava}
	Formalisoi lause. Onko lause tosi? Perustele.
	\begin{alakohdat}
	\alakohta{Jokainen kokonaisluku on joko positiivinen tai
	negatiivinen.}
	\alakohta{On olemassa sellainen kokonaisluku, jonka neliöjuuri on
	yhtä suuri kuin luku itse.}
	\alakohta{Minkään kokonaisluvun neliö ei ole $7$.}
	\end{alakohdat}%
	\begin{vastaus}
		\begin{alakohdat}
			\alakohta{$\forall x \in \mathbb{R}(x \ne 0)$ Lause on epätosi, sillä $0 \in \mathbb{R}$.}
			\alakohta{$\exists x \in \mathbb{Z}(\sqrt{x} = x)$ Lause on tosi, sillä $1 \in \mathbb{Z}$ ja $\sqrt{1} = 1$.}
			\alakohta{$\forall x \in \mathbb{Z}(x^2 \ne 7)$ Lause on tosi, sillä yhtälön $x^2 = 7$ ratkaisut $\sqrt{7}$ ja $-\sqrt{7}$ eivät ole kokonaislukuja.}
		\end{alakohdat}
	\end{vastaus}
\end{tehtava}

\end{tehtava}

\begin{tehtava}

	\begin{alakohdat}
	\alakohta{$\forall x \exists y P(x, y)$}
	\alakohta{$\exists x \forall y P(x, y)$}
	\alakohta{$\exists y \forall x P(x, y)$}

	\begin{vastaus}

			\alakohta{Jokaisella on jonkun puhelinnumero.}
			\alakohta{Ainakin yhdellä pelaajalla on kaikkien puhelinnumerot.}
			\alakohta{Ainakin yhden pelaajan puhelinnumero on kaikilla pelaajilla.}
		\end{alakohdat}
	\end{vastaus}
\end{tehtava}

\begin{tehtava}

\begin{tehtava}
	Olkoon perusjoukko $xy$-tason suorien joukko. Olkoot
	lauseet $N(x)$: ''suora on nouseva'' ja $L(x)$: ''suora on laskeva''.
	Suomenna lause. Onko lause tosi?
	\begin{alakohdat}
	\alakohta{$\exists x L(x)$}
	\alakohta{$\forall x (N(x) \lor L(x))$}
	\alakohta{$\exists x (\lnot N(x) \land \lnot L(x))$}
	\alakohta{$\forall x \lnot N(x)$}
	\end{alakohdat}%
	\begin{vastaus}
		\begin{alakohdat}
			\alakohta{On olemassa suora, joka on laskeva. Lause on tosi.}
			\alakohta{Kaikki suorat ovat joko nousevia tai laskevia. Lause on epätosi.}
			\alakohta{On olemassa suora, joka ei ole nouseva eikä laskeva. Lause on tosi.}
			\alakohta{Mikään suora ei ole nouseva.}
		\end{alakohdat}
	\end{vastaus}
\end{tehtava}

\begin{tehtava}
	Onko lause tosi? Perustele.
	\begin{alakohdat}
	\alakohta{$\forall x\in\rr (x^2 \ge 0)$}
	\alakohta{$\exists x\in\nn (x^2 = -9)$}
	\alakohta{$\forall x\in\rr ((x-1)(x-3) \ge 0)$}
	\alakohta{$\exists x\in\zz (-x^2 - 2 \le 0)$}
	\end{alakohdat}%
	\begin{vastaus}
		\begin{alakohdat}
			\alakohta{Lause on tosi, sillä reaaliluvun neliö ei ole koskaan negatiivinen.}
			\alakohta{Lause on epätosi, sillä $\forall x \in \mathbb{N}(x^2 \ge 0)$.}
			\alakohta{Lause on epätosi, sillä $2 \in \mathbb{R}$ ja $(2 - 1)(2 - 3) = -1 \cdot 1 = -1 < 0$.}
			\alakohta{Lause on tosi, sillä $0 \in \mathbb{R}$ ja $-0^2 - 2 = -2 \le 0$.}
		\end{alakohdat}
	\end{vastaus}
\end{tehtava}

\begin{tehtava}
	Onko lause tosi? Perustele.
	\begin{alakohdat}
	\alakohta{$\exists x\in \rr (x^2 - 3x + 3 = 0)$}
	\alakohta{$\forall x\in \rr (x^2 - 3x + 3 \ge 0)$}
	\end{alakohdat}%
	\begin{vastaus}
		\begin{alakohdat}
			\alakohta{Ratkaistaan funktion $x^2 - 3x + 3 = 0$ nollakohdat:
			$$x = \frac{-(-3) \pm \sqrt{(-3)^2 - 4 \cdot 1 \cdot 3}}{2 \cdot 1} = \frac{3 \pm \sqrt{-3}}{2}$$
			Koska diskriminantti $-3$ on negatiivinen, niin yhtälöllä ei ole ratkaisuja. Lisäksi koska yhtälön $x^2 - 3x + 3$ kuvaaja on ylöspäin aukeava paraabeli, niin pätee $x^2 - 3x + 3 > 0$ $\forall x \in \mathbb{R}$. Siis lause on epätosi.}
			\alakohta{Koska $x^2 - 3x + 3 > 0$ $\forall x \in \mathbb{R}$, niin $\forall x \in \mathbb{R}(x^2 - 3x + 3 \ge 0)$. Siis lause on tosi.}
		\end{alakohdat}
	\end{vastaus}
\end{tehtava}

\begin{tehtava}
	\begin{alakohdat}
	\alakohta{Osoita, että yhtälö $\sqrt{xy} = \sqrt{x}\sqrt{y}$ ei
	pidä paikkaansa kaikilla reaaliluvuilla $x$ ja $y$.}
	\alakohta{Osoita, että on olemassa sellaiset reaaliluvut $x$ ja
	$y$, että yhtälö $\sqrt{xy} = \sqrt{x}\sqrt{y}$ pitää paikkansa.}
	\alakohta{Millä ehdolla yhtälö $\sqrt{xy} = \sqrt{x}\sqrt{y}$
	pitää yleisesti paikkansa?}
	\end{alakohdat}%
	\begin{vastaus}
		\begin{alakohdat}
			\alakohta{Jos $x = y = -1 \in \mathbb{R}$, niin $\sqrt{xy} = \sqrt{(-1) \cdot (-1)} = \sqrt{1} = 1$, mutta $\sqrt{x} \cdot \sqrt{y} = \sqrt{-1} \cdot \sqrt{-1}$, jota ei ole määritelty reaalilukujen joukossa.}
			\alakohta{Kun $x = y = 1 \in \mathbb{R}$, niin $\sqrt{xy} = \sqrt{1 \cdot 1} = \sqrt{1} = 1 = 1 \cdot 1 = \sqrt{1} \cdot \sqrt{1} = \sqrt{x} \cdot \sqrt{y}$.}
			\alakohta{$x \ge 0 \land y \ge 0$}
		\end{alakohdat}
	\end{vastaus}
\end{tehtava}

\begin{tehtava}

	\begin{alakohdat}

	\alakohta{$\exists y\in \rr \forall x\in \rr (x^2 - 4 > y)$}
	\end{alakohdat}%
	\begin{vastaus}
		\begin{alakohdat}
			\alakohta{Lause on tosi, sillä jos $x \in \mathbb{Z}_{+}$ ja $y = x^2 \in \mathbb{Z}_{+}$, niin $\sqrt{y} = \sqrt{x^2} = |x| = x$.}
			\alakohta{Koska $\forall x \in \mathbb{R}(x \ge 0 > -1)$, niin $\forall x \in \mathbb{R}(x > -1) \Leftrightarrow \forall x \in \mathbb{R}(x - 4 > -5)$. Valitaan $y = -5$. Nyt pätee $\exists y \in \mathbb{R} \forall x \in \mathbb{R}(x - 4 > y)$ eli lause on tosi.}

	\end{vastaus}

\begin{tehtava}
	Muodosta lauseen negaatio. Onko negaatio tosi?
	\begin{alakohdat}
	\alakohta{$\forall x\in ]1, \infty [ (\sqrt{x} < x)$}
	\alakohta{$\exists x\in \qq (x^3 = 5)$}
	\alakohta{$\forall n \in \zz \exists m \in \zz ((n = 2m) \lor (n =
	2m+1))$}
	\end{alakohdat}%
	\begin{vastaus}
		\begin{alakohdat}
			\alakohta{$\exists x \in {]-1, \infty[}(\sqrt{x} \ge x)$ Negaatio on epätosi, koska
			\begin{align*}
			& \sqrt{x} > 1 && \parallel \cdot \sqrt{x} > 0 \\
			\Leftrightarrow & \sqrt{x} \cdot \sqrt{x} > \sqrt{x} && \\
			\Leftrightarrow & x > \sqrt{x} \forall x \in {]1, \infty[} &&
			\end{align*}
			}
			\alakohta{$\forall x \in \mathbb{Q}(x^3 \ne 5)$ Negaatio on tosi, sillä yhtälön $x^3 = 5$ ainoa ratkaisu on $x = \sqrt[3]{5} \notin \mathbb{Q}$.}
			\alakohta{$\forall n \in \mathbb{Z} \forall m \in \mathbb{Z} ((n \ne 2m) \land (n \ne 2m + 1))$ Valitaan $n = 2 \in \mathbb{Z}$ ja $m = 1 \in \mathbb{Z}$. Tällöin $n = 2 = 2 \cdot 1 = 2m$, joten negaatio on epätosi.}
		\end{alakohdat}
	\end{vastaus}
\end{tehtava}

\begin{tehtava}
	Funktiota $f\colon X\to Y$ voidaan ajatella kahden
	muuttujan avoimena lauseena $P(x, y)$: ''$f(x) = y$'', missä
	$x$ kuuluu määrittelyjoukkoon $X$ ja $y$ maalijoukkoon $Y$.
	Funktiolta edellytetään lisäksi, että
	\begin{itemize}
	\item $\forall x \exists y P(x, y)$, ja
	\item $\lnot (\exists x \exists y \exists z (P(x, y) \land P(x, z)
	\land (y \neq z)))$.
	\end{itemize}
	Tulkitse sanallisesti tai kuvaa käyttäen, mitä tämä määritelmä
	tarkoittaa.%
	\begin{vastaus}
		\begin{alakohdat}
			\alakohta{Funktio $f : X \to Y$ saa jokaisessa pisteessä $x \in X$ jonkin arvon.}
			\alakohta{Funktio $f : X \to Y$ ei saa kahta eri arvoa missään pisteessä $x \in X$.}
		\end{alakohdat}
	\end{vastaus}
\end{tehtava}
